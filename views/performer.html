// GETS VOLUME - MODIFY ONLY MULTIPLIER
function getLevel(){
    if(songPlaying){
        return amp.getLevel()*10;
    } else {
        return mic.getLevel()*100;
    }
}

// DRAW LOOP
function draw() {

  let vol = getLevel();
  
  background(30);
  
	for(var i=0;i<100;i++){
    
    //Draw Outer Ellipses
    fill(30);
    stroke(240);
    ellipseRadius = (ellipseWidth%i)*4*vol;
    ellipse((windowWidth / 2), (windowHeight / 2), ellipseRadius, ellipseRadius);
    
  	//Draw Inner Ellipses
    stroke(255,0,0);
    ellipseRadius = (ellipseWidth%i)*3*vol;
    ellipse((windowWidth / 2), (windowHeight / 2), ellipseRadius, ellipseRadius);
  
  }

  ellipseWidth++;
  
  if(ellipseWidth>920){
    ellipseWidth=1;
  }
  
}
